class SpeechPlayer {
    constructor(options = {}) {
        this.defaultOptions = {
            volume: 1,
            rate: 0.8,
            pitch: 1,
            lang: 'zh-CN',
            voice: null
        };
        
        this.options = { ...this.defaultOptions, ...options };
        this.isPlaying = false;
        this.queue = [];
        this.currentUtterance = null;
        
        this.initVoices();
    }
    
    // ÂàùÂßãÂåñËØ≠Èü≥
    initVoices() {
        const setVoices = () => {
            const voices = window.speechSynthesis.getVoices();
            if (!this.options.voice && voices.length > 0) {
                // Ëá™Âä®ÈÄâÊã©ÂêàÈÄÇÁöÑ‰∏≠ÊñáËØ≠Èü≥
                const chineseVoice = voices.find(voice => 
                    voice.lang.includes('zh') && voice.localService
                ) || voices.find(voice => voice.lang.includes('zh'));
                
                if (chineseVoice) {
                    this.options.voice = chineseVoice;
                }
            }
        };
        
        setVoices();
        window.speechSynthesis.onvoiceschanged = setVoices;
    }
    
    // Êí≠Êä•ÊñáÊú¨
    speak(text, customOptions = {}) {
        if (!text || typeof text !== 'string') {
            console.warn('Êí≠Êä•ÊñáÊú¨‰∏çËÉΩ‰∏∫Á©∫');
            return Promise.reject(new Error('Invalid text'));
        }
        
        return new Promise((resolve, reject) => {
            // ÂÅúÊ≠¢ÂΩìÂâçÊí≠Êä•
            this.stop();
            
            const utterance = new SpeechSynthesisUtterance();
            const finalOptions = { ...this.options, ...customOptions };
            
            // ËÆæÁΩÆÂ±ûÊÄß
            utterance.text = text;
            utterance.volume = finalOptions.volume;
            utterance.rate = finalOptions.rate;
            utterance.pitch = finalOptions.pitch;
            utterance.lang = finalOptions.lang;
            
            if (finalOptions.voice) {
                utterance.voice = finalOptions.voice;
            }
            
            // ‰∫ã‰ª∂ÁõëÂê¨
            utterance.onstart = () => {
                this.isPlaying = true;
                console.log('ÂºÄÂßãÊí≠Êä•:', text);
            };
            
            utterance.onend = () => {
                this.isPlaying = false;
                this.currentUtterance = null;
                console.log('Êí≠Êä•ÂÆåÊàê:', text);
                resolve();
            };
            
            utterance.onerror = (event) => {
                this.isPlaying = false;
                this.currentUtterance = null;
                console.error('Êí≠Êä•ÈîôËØØ:', event.error);
                reject(new Error(event.error));
            };
            
            utterance.onpause = () => {
                console.log('Êí≠Êä•ÊöÇÂÅú');
            };
            
            utterance.onresume = () => {
                console.log('Êí≠Êä•ÊÅ¢Â§ç');
            };
            
            this.currentUtterance = utterance;
            window.speechSynthesis.speak(utterance);
        });
    }
    
    // ÂÅúÊ≠¢Êí≠Êä•
    stop() {
        if (this.isPlaying) {
            window.speechSynthesis.cancel();
            this.isPlaying = false;
            this.currentUtterance = null;
        }
    }
    
    // ÊöÇÂÅúÊí≠Êä•
    pause() {
        if (this.isPlaying && !window.speechSynthesis.paused) {
            window.speechSynthesis.pause();
        }
    }
    
    // ÊÅ¢Â§çÊí≠Êä•
    resume() {
        if (this.isPlaying && window.speechSynthesis.paused) {
            window.speechSynthesis.resume();
        }
    }
    
    // ÈòüÂàóÊí≠Êä•
    async speakQueue(textArray, delay = 500) {
        for (const text of textArray) {
            try {
                await this.speak(text);
                if (delay > 0) {
                    await this.delay(delay);
                }
            } catch (error) {
                console.error('ÈòüÂàóÊí≠Êä•ÈîôËØØ:', error);
            }
        }
    }
    
    // Âª∂ËøüÂáΩÊï∞
    delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
    
    // Ëé∑ÂèñÁä∂ÊÄÅ
    getStatus() {
        return {
            isPlaying: this.isPlaying,
            isPaused: window.speechSynthesis.paused,
            hasPending: window.speechSynthesis.pending
        };
    }
    
    // ËÆæÁΩÆÈªòËÆ§ÈÄâÈ°π
    setDefaultOptions(options) {
        this.options = { ...this.options, ...options };
    }
}

    // Ê£ÄÊü•Áî®Êà∑ÂÅèÂ•Ω
    checkAccessibilityPreference() {
        return localStorage.getItem('speech-enabled') === 'true' ||
               window.matchMedia('(prefers-reduced-motion: no-preference)').matches;
    }
    
    // ËÆæÁΩÆÈîÆÁõòÂø´Êç∑ÈîÆ
    setupKeyboardShortcuts() {
        document.addEventListener('keydown', (event) => {
            // Ctrl + Shift + S ÂàáÊç¢ËØ≠Èü≥Êí≠Êä•
            if (event.ctrlKey && event.shiftKey && event.key === 'S') {
                this.toggle();
                event.preventDefault();

            // Ctrl + Shift + Space ÂÅúÊ≠¢Êí≠Êä•
            if (event.ctrlKey && event.shiftKey && event.code === 'Space') {
                this.stop();
                event.preventDefault();
            }

    }

        // Ê†πÊçÆÂÖÉÁ¥†Á±ªÂûãÁîüÊàêÊí≠Êä•ÊñáÊú¨
        switch (element.tagName.toLowerCase()) {
            case 'button':
                text = `ÊåâÈíÆÔºö${element.textContent || element.getAttribute('aria-label')}`;
                break;
            case 'input':
                const inputType = element.type || 'text';
                const label = element.getAttribute('aria-label') || element.placeholder;
                text = `${inputType}ËæìÂÖ•Ê°ÜÔºö${label}`;
                break;
            case 'a':
                text = `ÈìæÊé•Ôºö${element.textContent}`;
                break;
            case 'h1':
            case 'h2':
            case 'h3':
            case 'h4':
            case 'h5':
            case 'h6':
                text = `${element.tagName}Ê†áÈ¢òÔºö${element.textContent}`;
                break;
            default:
                text = element.textContent || element.getAttribute('aria-label');
        }
        
        if (text) {
            this.speechPlayer.speak(text);
        }
    }
    
    // Êí≠Êä•Ë°®ÂçïÈ™åËØÅÁªìÊûú

        const text = `${field}${status}${message ? 'Ôºö' + message : ''}`;

            pitch: isValid ? 1 : 1.3
        });
    }
    
    // ÂàáÊç¢ËØ≠Èü≥Êí≠Êä•
    toggle() {

        }
    }
    
    // ÂÅúÊ≠¢Êí≠Êä•
    stop() {
        this.speechPlayer.stop();
    }
